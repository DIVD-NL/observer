          $(document).ready(function () {
              var errorReporter = new ErrorReporter('#loginForm', {});
              $('#loginForm').submit(function (event) {
                  event.preventDefault();
                  var $this = $(this);
                  errorReporter.clearErrors();
                  errorReporter.validateNamedField('email', /.+@.+/, 'Mandatory');
                  errorReporter.validateNamedField('password', /.+/, 'Mandatory');
                  if (errorReporter.isError()) {
                      return;
                  }
                  var reqData = {
                      email : errorReporter.getNamedFieldValue('email'),
                      password : errorReporter.getNamedFieldValue('password'),
                      next : errorReporter.getNamedFieldValue('next')
                  };
                  $this.find('button[type="submit"]').busy();
                  errorReporter.postJSON('/dologin', reqData, function (data) {
                      $this.find('button[type="submit"]').busy('hide');
                      window.location = '' + data.target;
                  }).complete(function () {
                      $this.find('button[type="submit"]').busy('hide');
                  });
              });
          });
      </script>

                e.stopPropagation();
                e.preventDefault();
            });
            $(document).on('dragleave', function (e) {
                e.stopPropagation();

            });
            $(document).on('drop', function (e) {
                e.stopPropagation();
                e.preventDefault();
            });
            // load the customer info
            var customerCode = '';
            if (customerCode === '') {
                var hash = window.location.hash;
                if (hash !== null && hash.length !== 0 && hash.length !== 1) {
                    var parts = hash.substr(1).split('/');
                    if (parts.length >= 2) {
                        customerCode = parts[1];
                    }
                }
            }
            if (customerCode !== '') {
                var errorReporter = new ErrorReporter('body', {});
                var reqData = {
                    customerCode : customerCode
                };
                errorReporter.getJSON('/app/customer/selectCustomer', reqData, function (data) {
                    var name = $('<div></div>').html(data.name).text();
                    var customerCode = $('<div></div>').html(data.customerCode).text();
                    $('[data-text="customerInfo"]').html(name + ' / ' + customerCode);
                });
            }
        });
