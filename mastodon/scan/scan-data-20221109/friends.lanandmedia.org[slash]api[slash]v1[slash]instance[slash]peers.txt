	var tmpStr = $("#comment-edit-text-" + id).val();

		$("#comment-edit-text-" + id).addClass("comment-edit-text-full");

		textarea.focus();

		selected.text = "["+BBcode+"]" + selected.text + "[/"+BBcode+"]";

	return true;
}
