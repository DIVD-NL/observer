      return s;
    }
  
    function scriptLoaderLoadScript(scriptContent) {
      scriptContent = decodeURIComponent(scriptContent);
  
      var temporaryContainer = document.createElement('div');
      temporaryContainer.innerHTML = scriptContent;
  
      var temporaryScripts = Array.prototype.slice.call(temporaryContainer.querySelectorAll('script'));
      temporaryScripts.forEach(function loadScript(script) {
        try {
          if (script.textContent) eval(script.textContent); 
          if (script.src) {
            script = downloadScript(script); 
          }
          head.appendChild(script);
        } catch(error) {
          console.error(error);
        }
      });
    }
  
    function scriptLoaderLoadScripts(scripts) {
      if (
        !scripts
        || !Array.isArray(scripts)
        || !scripts.filter(function(s) { return s; }).length
      ) return;
  
      scripts.forEach(function(scriptConfig) {
        setTimeout(function() { scriptLoaderLoadScript(scriptConfig); }, 0);
      });
    }
  
    window.addEventListener('load', function scriptLoaderLoad() {
  
      scriptLoaderLoadScripts(window.scriptsToLoad);
      console.info('Finish - encoded scripts')
    });
